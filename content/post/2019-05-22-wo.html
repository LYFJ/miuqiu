library(maps)
library(mapdata)
map("world", fill = TRUE, col = rainbow(200),ylim = c(-20, 90), mar = c(0, 0, 0, 0))
title("世界地图")

library(maps)
library(mapdata)
map("china", col = "red4", ylim = c(17, 56), panel.first = grid())
title("中国地图")
