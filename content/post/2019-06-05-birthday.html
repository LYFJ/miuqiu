birthday <-function(){
  x<-Sys.Date()
  y<-as.Date("1998-06-29")
  print(paste("今天我出生",x-y,"天"))
}
birthday()
## [1] "今天我出生 7647 天"
birthday <-function(){
  x<-as.Date("1967-11-06")
  y<-as.Date("1998-06-29")
  print(paste("我和我妈相差",y-x,"天"))
}
birthday()
## [1] "我和我妈相差 11193 天"
birthday <-function(){
  x<-Sys.Date()
  y<-as.Date("2019-10-1")
  z<-as.Date("2019-09-13")
  print(paste("距离国庆放假还有",y-x,"天"))
  print(paste("距离中秋节放假还有",z-x,"天"))
  print(paste("国庆与中秋相差",y-z,"天"))
}
birthday()
## [1] "距离国庆放假还有 117 天"
## [1] "距离中秋节放假还有 99 天"
## [1] "国庆与中秋相差 18 天"
