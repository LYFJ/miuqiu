birthday <-function(){
  x<-Sys.Date()
  y<-as.Date("1998-06-29")
  print(paste("今天我出生多少天",x-y,"天"))
}
birthday()
## [1] "今天我出生多少天 7647 天"
birthday <-function(){
  x<-as.Date("1967-11-06")
  y<-as.Date("1998-06-29")
  print(paste("我和我妈相差多少岁",y-x,"天"))
}
birthday()
## [1] "我和我妈相差多少岁 11193 天"
birthday <-function(){
  x<-Sys.Date()
  y<-as.Date("2019-10-1")
  z<-as.Date("2019-09-13")
  print(paste("距离国庆放假还有多少天",y-x,"天"))
  print(paste("距离中秋节放假还有多少天",z-x,"天"))
}
birthday()
